	$git init

	Customize Git
	
	$git config --global user.name "Dug Wilson"
	$git config --global user.email "dugald.wilson@gmail.com"
	
	$git config --global color.status auto
	$git config --global color.branch auto
	$git config --global color.ui always
	
	Add files
	
	$git add .
